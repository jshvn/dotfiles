    The runner needs to handle three distinct fixture types — each requires running its corresponding lint check against ONLY the fixture file (not the whole repo). Implement as a single cmds: block (S9 envelope) that switches on the fixture directory name prefix:

    ```
    {{.DOTFILES_MESSAGES}}
    failures=0
    fixtures_dir="taskfiles/test/lint-fixtures"
    for fix in "$fixtures_dir"/[0-9]*-*/; do
      name=$(basename "$fix")
      expect=$(head -n1 "$fix/expect" | tr -d '[:space:]')
      case "$name" in
        02*|03a-*)
          # LINT-02 / LINT-03a: scope to fixture's Taskfile.yml
          # Run the lint:taskfile detection logic against just this file.
          # Use the same pipeline as taskfiles/lint.yml's task `taskfile`,
          # but with TASKFILE_GLOB overridden to the fixture file.
          if (cd "$fix" && yq '.tasks[] | select(.status) | .status' Taskfile.yml 2>/dev/null \
              | ggrep -nE '\\$[A-Za-z_][A-Za-z0-9_]*' \
              | ggrep -vE '\\$\\(' \
              | ggrep -vE '\\{\\{' >/dev/null) \
             || (cd "$fix" && yq '.tasks | to_entries | .[] | select(.value | has("cmds")) | select(.value | has("status") | not) | .key' Taskfile.yml 2>/dev/null | grep -q .); then
            actual=fail
          else
            actual=pass
          fi
          ;;
        03b-*)
          # LINT-03b: bare ln -s. Bind helper-allowed by filename.
          file=$(ls "$fix"/*.yml | head -1)
          base=$(basename "$file")
          if ggrep -qE '\\bln[[:space:]]+-s' "$file" && [[ "$base" != "helpers.yml" ]]; then
            actual=fail
          else
            actual=pass
          fi
          ;;
        04*)
          # LINT-04: executable .zsh missing set -euo pipefail
          file=$(ls "$fix"/*.zsh | head -1)
          if [[ -x "$file" ]] && ! head -30 "$file" | ggrep -qE '^set -euo pipefail$'; then
            actual=fail
          else
            actual=pass
          fi
          ;;
        05*)
          # LINT-05: portability pattern (warn-only). actual="warn" if pattern found.
          file=$(ls "$fix"/*.zsh | head -1)
          if ggrep -qE '\\b(pbcopy|pbpaste|osascript|sw_vers|dscl|PlistBuddy)\\b|defaults (read|write)' "$file"; then
            actual=warn
          else
            actual=pass
          fi
          ;;
        07*)
          # LINT-07: zsh -n parse error
          file=$(ls "$fix"/*.zsh | head -1)
          if zsh -n "$file" 2>/dev/null; then
            actual=pass
          else
            actual=fail
          fi
          ;;
        *) actual=unknown ;;
      esac
      if [[ "$actual" == "$expect" ]]; then
        check "${name}: expect=${expect} actual=${actual}"
      else
        cross "${name}: expect=${expect} actual=${actual}"
        failures=$((failures + 1))
      fi
    done
    if [[ $failures -gt 0 ]]; then
      error "lint:test-fixtures: $failures fixture(s) failed expected behavior"
    else
      success "lint:test-fixtures: all $(ls -d "$fixtures_dir"/[0-9]*-*/ | wc -l | tr -d ' ') fixtures behaved as expected"
    fi
    exit "$failures"
    ```
