<!-- D-13 banner-parity lint check sketch (LINT-08 reused -- the rule number was deprecated and is free to reuse): -->

  # lint-allow: cmds-without-status
  banner-parity:
    desc: "Banner parity check: every non-internal top-level task appears in default:'s cmd block (LINT-08 -- D-13)"
    internal: true
    cmds:
      - |
        # The default: task's cmd block must list every non-internal
        # top-level task (no `:` in the name). Drift detection:
        # diff yq-extracted task list against the banner cmd text.
        {{.DOTFILES_MESSAGES}}
        failures=0
        ROOT_TASKFILE="{{.DOTFILEDIR}}/Taskfile.yml"
        public_top_tasks=$(yq '.tasks | to_entries | .[] | select(.value.internal // false | not) | .key | select(test("^[^:]+$"))' "$ROOT_TASKFILE")
        banner_body=$(yq '.tasks.default.cmds[0]' "$ROOT_TASKFILE")
        for task_name in $public_top_tasks; do
          # Skip the `default` task itself -- it cannot reference itself in its own banner.
          [[ "$task_name" == "default" ]] && continue
          if ! echo "$banner_body" | ggrep -qF "$task_name"; then
            cross "LINT-08: 'task $task_name' is public but missing from default:'s banner -- update Taskfile.yml"
            failures=$(( failures + 1 ))
          else
            check "LINT-08: $task_name in banner"
          fi
        done
        exit "$failures"
